##### 1.1../db/mysql文件

```js
const mysql = require('mysql') // 导入mysql模块

const { MYSQL_CONF } = require('../conf/db') // 导入数据库配置对象

// 创建连接对象
const con = mysql.createConnection(MYSQL_CONF)

// 开始连接
con.connect()

// 统一执行 SQL 的函数
function exec(sql) {
    const promise = new Promise((resolve, reject) => {
        // 执行SQL查询
        con.query(sql, (err, result) => {
            if (err) {
                // 查询出错，reject返回错误信息
                reject(err)
                return
            }
            // 查询成功，resolve返回查询结果
            resolve(result)
        })
    })
    return promise
}


module.exports = {
    exec,
    escape: mysql.escape
}
```

#### 	 2. user.js

```js
var express = require('express');
var router = express.Router();
const { login, register } = require('../controller/user.js')
const { SuccessModel, ErrorModel } = require('../model/resModel.js')
// 登录
router.post('/login', function (req, res, next) {
  const { username, password } = req.body
  const result = login(username, password)
  return result.then(data => {
    if (data.username) {
      // 设置session
      req.session.username = data.username
      req.session.realname = data.realname
      res.json(
        new SuccessModel('登录成功')
      )
      return
    }
    res.json(
      new ErrorModel('登录失败')
    )
  })
});

// 注册
router.post('/register', function (req, res, next) {
  const { username, password } = req.body
  const result = register(username, password)
  return result.then(val => {
    if (val){
      res.json(
        new SuccessModel(val,'注册成功')
      )}
      else {
        res.json(
          new ErrorModel('账号已存在')
        )
    }
  })
});

module.exports = router;
```

### 5. resModel（class类，还不太懂，但是基本知道有什么作用）

##### (注释使用chatgpt写的)

```js
// 定义基础模型类
class BaseModel {
  // 构造函数，接收数据和消息作为参数
  constructor(data, message) {
    // 如果数据为字符串类型，则将其作为消息处理
    if (typeof data === 'string') {
      // 将数据赋值给消息属性
      this.message = data;
      // 清空数据和消息
      data = null;
      message = null;
    }
    
    // 如果数据存在，将其赋值给数据属性
    if (data) {
      this.data = data;
    }

    // 如果消息存在，将其赋值给消息属性
    if (message) {
      this.message = message;
    }
  }
}

// 成功模型类，继承自基础模型类
class SuccessModel extends BaseModel {
  // 构造函数，接收数据和消息作为参数
  constructor(data, message) {
    // 调用父类的构造函数，传递数据和消息参数
    super(data, message);
    // 设置错误码为0，表示成功
    this.errno = 0;
  }
}

// 错误模型类，继承自基础模型类
class ErrorModel extends BaseModel {
  // 构造函数，接收数据和消息作为参数
  constructor(data, message) {
    // 调用父类的构造函数，传递数据和消息参数
    super(data, message);
    // 设置错误码为-1，表示错误
    this.errno = -1;
  }
}

// 导出成功模型类和错误模型类
module.exports = {
  SuccessModel,
  ErrorModel
};
```
